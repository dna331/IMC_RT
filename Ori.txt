import os
import requests
from bs4 import BeautifulSoup
import sys

def get_wikipedia_page_url(title):
    base_url = "https://vi.wikipedia.org/wiki/"
    return base_url + title.replace(" ", "_")


def get_image_and_caption(page_url):
    response = requests.get(page_url)
    soup = BeautifulSoup(response.text, 'html.parser')

    images = []
    for figure in soup.find_all('figure', class_=['mw-default-size','mw-halign-right']):
        img = figure.find('img', src=True)
        image_url = 'https:' + img['src']
        
        caption = figure.find('figcaption')
        caption_text = caption.get_text(strip=True) if caption else None

        images.append({'url': image_url, 'caption': caption_text})
    for div in soup.find_all('div', class_=['trow']):
        img = div.find('img', src=True)
        
        if img:  # Check if the img object exists before accessing its attributes
            image_url = 'https:' + img['src']

            caption = div.find('div', class_='thumbcaption')
            
            caption_text = caption.get_text(strip=True) if caption else None

            images.append({'url': image_url, 'caption': caption_text})

    return images



def download_image(url, file_path):
    response = requests.get(url)
    with open(file_path, 'wb') as file:
        file.write(response.content)

def main(page_title):
    # page_title = "Bóng đá"  # You can replace this with any other Wikipedia page title
    # page_url = get_wikipedia_page_url(page_title)
    page_url = get_wikipedia_page_url(page_title)

    images = get_image_and_caption(page_url)
    download_directory = page_title

    if not os.path.exists(download_directory):
        os.makedirs(download_directory)

    for idx, image in enumerate(images, start=1):
        file_name = f"{page_title}_{idx}.jpg"
        file_path = os.path.join(download_directory, file_name)

        download_image(image['url'], file_path)
        print(f"Downloaded image {idx}: {file_path}")

        caption_file_name = f"{page_title}_{idx}_caption.txt"
        caption_file_path = os.path.join(download_directory, caption_file_name)

        if image['caption']:
            with open(caption_file_path, 'w', encoding='utf-8') as caption_file:
                caption_file.write(image['caption'])
            print(f"Downloaded caption {idx}: {caption_file_path}")
        else:
            print(f"No caption for image {idx}")

        print()

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Please provide a Wikipedia page title as an argument.")
    else:
        page_title = " ".join(sys.argv[1:])
        main(page_title)
